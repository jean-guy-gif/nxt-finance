# V3.1 — Fondations & Infrastructure — Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Poser les bases techniques partagées par tous les modules V3 : tables transverses, job-orchestrator, llm-gateway, registries, storage, Edge Functions.

**Architecture:** Deux tables transverses (processing_jobs, llm_generations) pilotent tous les traitements longs et la traçabilité IA. Le job-orchestrator (Next.js server) orchestre le cycle de vie des jobs. Le llm-gateway (Edge Function) isole les appels Claude API. Les registries versionnés (fichiers config) centralisent benchmarks, seuils et prompts.

**Tech Stack:** Next.js 16.1 + Supabase (PostgreSQL, Edge Functions, Storage) + Claude API + React Query 5 + Zustand 5 + TypeScript strict

-- ============================================
-- NXT Finance V3 — Foundation tables

-- ---- ENUMS ----

CREATE TYPE job_type AS ENUM (
  'bilan_parsing',
  'analysis_generation',
  'bp_generation',
  'dossier_generation',
  'dossier_export',
  'llm_generation'
);

CREATE TYPE job_status AS ENUM (
  'queued',
  'processing',
  'completed',
  'failed',
  'cancelled'
);

CREATE TYPE llm_output_type AS ENUM (
  'financial_insight',
  'bp_narrative',
  'alert_recommendation',
  'slide_narrative',
  'director_summary'
);

CREATE TYPE llm_generation_status AS ENUM (
  'pending',
  'completed',
  'failed'
);

-- ---- PROCESSING JOBS ----

CREATE TABLE processing_jobs (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  agency_id UUID NOT NULL REFERENCES agencies(id) ON DELETE CASCADE,
  job_type job_type NOT NULL,
  status job_status NOT NULL DEFAULT 'queued',
  related_type TEXT NOT NULL,
  related_id UUID NOT NULL,
  progress INTEGER DEFAULT 0,
  error_message TEXT,
  triggered_by UUID REFERENCES user_profiles(id),
  payload_json JSONB,
  started_at TIMESTAMPTZ,
  completed_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now(),

  CONSTRAINT chk_jobs_progress_range CHECK (progress >= 0 AND progress <= 100)
);

-- Prevent duplicate active jobs for the same target
CREATE UNIQUE INDEX idx_jobs_no_duplicate_active
  ON processing_jobs (related_type, related_id, job_type)
  WHERE status IN ('queued', 'processing');

-- Fast lookups
CREATE INDEX idx_jobs_agency ON processing_jobs (agency_id);
CREATE INDEX idx_jobs_status ON processing_jobs (status) WHERE status IN ('queued', 'processing');
CREATE INDEX idx_jobs_related ON processing_jobs (related_type, related_id);

-- ---- LLM GENERATIONS ----

CREATE TABLE llm_generations (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  agency_id UUID NOT NULL REFERENCES agencies(id) ON DELETE CASCADE,
  provider TEXT NOT NULL DEFAULT 'anthropic',
  model TEXT NOT NULL,
  prompt_version TEXT NOT NULL,
  input_refs JSONB NOT NULL DEFAULT '{}',
  output_type llm_output_type NOT NULL,
  output_id UUID,
  tokens_input INTEGER,
  tokens_output INTEGER,
  duration_ms INTEGER,
  status llm_generation_status NOT NULL DEFAULT 'pending',
  error_message TEXT,
  generated_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),

  CONSTRAINT chk_llm_tokens_positive CHECK (
    (tokens_input IS NULL OR tokens_input >= 0)
    AND (tokens_output IS NULL OR tokens_output >= 0)
  )
);

CREATE INDEX idx_llm_agency ON llm_generations (agency_id);
CREATE INDEX idx_llm_output ON llm_generations (output_type, output_id);
CREATE INDEX idx_llm_status ON llm_generations (status) WHERE status = 'pending';

-- ---- RLS ----

ALTER TABLE processing_jobs ENABLE ROW LEVEL SECURITY;
ALTER TABLE llm_generations ENABLE ROW LEVEL SECURITY;

-- Members can read jobs/generations for their agencies
CREATE POLICY "Agency members can read processing jobs"

  key: string,
  value: number
): 'healthy' | 'warning' | 'critical' {
  const benchmark = getBenchmarkForRatio(key);

  version?: string

  let resolved = template;
  for (const [key, value] of Object.entries(variables)) {
    resolved = resolved.replaceAll(`{${key}}`, value);
  }
  return resolved;
}

/**
 * Create a processing job with idempotency.
 * If an active job (queued/processing) already exists for the same target,
 * returns the existing job instead of creating a duplicate.
 */
export async function createJob(
  supabase: SupabaseClient,
  input: CreateJobInput
): Promise<JobResult> {
  // Check for existing active job (idempotency)
  const { data: existing } = await supabase
    .from('processing_jobs')
    .select('*')
    .eq('related_type', input.relatedType)
    .eq('related_id', input.relatedId)
    .eq('job_type', input.jobType)
    .in('status', ['queued', 'processing'])
    .maybeSingle();

  if (existing) {
    return { job: existing as ProcessingJob, isExisting: true };
  }

  const { data, error } = await supabase
    .from('processing_jobs')
    .insert({
      agency_id: input.agencyId,
      job_type: input.jobType,
      status: 'queued',
      related_type: input.relatedType,
      related_id: input.relatedId,
      payload_json: input.payload ?? null,
      triggered_by: input.triggeredBy ?? null,
      progress: 0,
    })
    .select('*')
    .single();

  if (error) throw error;
  return { job: data as ProcessingJob, isExisting: false };
}

// ============================================
// NXT Finance V3 — LLM Gateway (client-side orchestration)
// ============================================
// Orchestre les appels LLM via Edge Function.
// Persiste la traçabilité dans llm_generations.
// Dégradation gracieuse : retourne null si LLM indisponible.
// ============================================

import type { SupabaseClient } from '@supabase/supabase-js';
import type { LlmGeneration } from '@/types/models';
import type { LlmOutputType } from '@/types/enums';
import { DEFAULT_LLM_MODEL, DEFAULT_LLM_PROVIDER } from '@/lib/constants';
import { getPromptTemplate, resolvePromptVariables } from '@/lib/registries/prompt-registry';

export interface LlmRequest {
  type: LlmOutputType;
  promptVersion?: string;
  variables: Record<string, string>;
  agencyId: string;
  outputId?: string;
}

export interface LlmResponse {
  content: string;
  generationId: string;
}

/**
 * Generate content via LLM.
 * Creates a llm_generations record for traceability.
 * Returns null on failure (graceful degradation).
 */
export async function generateContent(
  supabase: SupabaseClient,
  request: LlmRequest
): Promise<LlmResponse | null> {
  const template = getPromptTemplate(request.type, request.promptVersion);
  if (!template) {
    console.error(`No prompt template found for type=${request.type} version=${request.promptVersion}`);
    return null;
  }

  // Create traceability record (pending)
  const { data: generation, error: insertError } = await supabase
    .from('llm_generations')
    .insert({
      agency_id: request.agencyId,
      provider: DEFAULT_LLM_PROVIDER,
      model: DEFAULT_LLM_MODEL,
      prompt_version: template.version,
      input_refs: request.variables,
      output_type: request.type,
      output_id: request.outputId ?? null,
      status: 'pending',
    })
    .select('*')
    .single();

  if (insertError || !generation) {
    console.error('Failed to create llm_generation record:', insertError?.message);
    return null;
  }

  const generationId = generation.id;
  const startTime = Date.now();

  try {
    // Resolve prompt
    const userPrompt = resolvePromptVariables(template.userPromptTemplate, request.variables);

    // Call Edge Function
    const { data: fnData, error: fnError } = await supabase.functions.invoke('llm-gateway', {
      body: {
        systemPrompt: template.systemPrompt,
        userPrompt,
        maxTokens: template.maxTokens,
        temperature: template.temperature,
        model: DEFAULT_LLM_MODEL,
      },
    });

    if (fnError) throw fnError;

    const duration = Date.now() - startTime;
    const content = fnData?.content ?? '';
    const tokensInput = fnData?.usage?.input_tokens ?? null;
    const tokensOutput = fnData?.usage?.output_tokens ?? null;

    // Update generation record (completed)
    await supabase
      .from('llm_generations')
      .update({
        status: 'completed',
        tokens_input: tokensInput,
        tokens_output: tokensOutput,
        duration_ms: duration,
        generated_at: new Date().toISOString(),
      })
      .eq('id', generationId);

    return { content, generationId };
  } catch (err) {
    const duration = Date.now() - startTime;
    const errorMessage = err instanceof Error ? err.message : 'Unknown LLM error';

    // Update generation record (failed)
    await supabase
      .from('llm_generations')
      .update({
        status: 'failed',
        error_message: errorMessage,
        duration_ms: duration,
      })
      .eq('id', generationId);

    console.error('LLM generation failed:', errorMessage);
    return null; // Graceful degradation
  }
}

// supabase/functions/llm-gateway/index.ts
// ============================================
// NXT Finance V3 — LLM Gateway Edge Function
// ============================================
// Isolates Claude API calls from the main Next.js runtime.
// Receives resolved prompts, calls Claude, returns content.
// Never calculates. Never accesses DB directly.
// ============================================

import "jsr:@supabase/functions-js/edge-runtime.d.ts";

const ANTHROPIC_API_URL = 'https://api.anthropic.com/v1/messages';

interface LlmRequestBody {
  systemPrompt: string;
  userPrompt: string;
  maxTokens: number;
  temperature: number;
  model: string;
}

Deno.serve(async (req: Request) => {
  // CORS headers for Supabase client calls
  if (req.method === 'OPTIONS') {
    return new Response('ok', {
      headers: {
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Methods': 'POST, OPTIONS',
        'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
      },
    });
  }

  try {
    const body: LlmRequestBody = await req.json();
    const { systemPrompt, userPrompt, maxTokens, temperature, model } = body;

    if (!systemPrompt || !userPrompt) {
      return new Response(
        JSON.stringify({ error: 'Missing systemPrompt or userPrompt' }),
        { status: 400, headers: { 'Content-Type': 'application/json' } }
      );
    }

    const apiKey = Deno.env.get('ANTHROPIC_API_KEY');
    if (!apiKey) {
      return new Response(
        JSON.stringify({ error: 'ANTHROPIC_API_KEY not configured' }),
        { status: 500, headers: { 'Content-Type': 'application/json' } }
      );
    }

    const response = await fetch(ANTHROPIC_API_URL, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': apiKey,
        'anthropic-version': '2023-06-01',
      },
      body: JSON.stringify({
        model: model || 'claude-sonnet-4-6',
        max_tokens: maxTokens || 500,
        temperature: temperature ?? 0.3,
        system: systemPrompt,
        messages: [
          { role: 'user', content: userPrompt },
        ],
      }),
    });

    if (!response.ok) {
      const errorText = await response.text();
      console.error('Claude API error:', response.status, errorText);
      return new Response(
        JSON.stringify({ error: `Claude API error: ${response.status}`, details: errorText }),
        { status: 502, headers: { 'Content-Type': 'application/json' } }
      );
    }

    const result = await response.json();
    const content = result.content?.[0]?.text ?? '';
    const usage = result.usage ?? {};

    return new Response(
      JSON.stringify({
        content,
        usage: {
          input_tokens: usage.input_tokens ?? null,
          output_tokens: usage.output_tokens ?? null,
        },
      }),
      { status: 200, headers: { 'Content-Type': 'application/json' } }
    );
  } catch (err) {
    console.error('LLM Gateway error:', err);
    return new Response(
      JSON.stringify({ error: err instanceof Error ? err.message : 'Internal error' }),
      { status: 500, headers: { 'Content-Type': 'application/json' } }
    );
  }
});

```typescript
// src/features/shared/services/__tests__/job-orchestrator.test.ts
// ============================================
// NXT Finance V3.1 — Integration test script
// ============================================
// Run this in a server action or API route to validate:
// 1. Job creation with idempotency
// 2. Job lifecycle (queued → processing → completed)
// 3. Duplicate prevention
// 4. LLM gateway call + traceability
// ============================================

import type { SupabaseClient } from '@supabase/supabase-js';
import {
  createJob,
  startJob,
  updateJobProgress,
  completeJob,
  failJob,
  fetchJob,
  canRetryJob,
} from '../job-orchestrator';
import { generateContent } from '../llm-gateway';

interface TestResult {
  name: string;
  passed: boolean;
  details: string;
}

export async function runV31IntegrationTests(
  supabase: SupabaseClient,
  agencyId: string
): Promise<TestResult[]> {
  const results: TestResult[] = [];
  const testRelatedId = crypto.randomUUID();

  // Test 1: Create a job
  try {
    const { job, isExisting } = await createJob(supabase, {
      agencyId,
      jobType: 'llm_generation',
      relatedType: 'test',
      relatedId: testRelatedId,
      payload: { test: true },
    });

    results.push({
      name: 'Job creation',
      passed: job.status === 'queued' && !isExisting,
      details: `Job ${job.id} created with status=${job.status}`,
    });

    // Test 2: Idempotency — creating same job returns existing
    const { job: duplicateJob, isExisting: isDuplicate } = await createJob(supabase, {
      agencyId,
      jobType: 'llm_generation',
      relatedType: 'test',
      relatedId: testRelatedId,
    });

    results.push({
      name: 'Job idempotency',
      passed: isDuplicate && duplicateJob.id === job.id,
      details: `Duplicate check: isExisting=${isDuplicate}, sameId=${duplicateJob.id === job.id}`,
    });

    // Test 3: Start the job (logical lock)
    const startedJob = await startJob(supabase, job.id);

    results.push({
      name: 'Job start with lock',
      passed: startedJob.status === 'processing' && startedJob.started_at !== null,
      details: `Status=${startedJob.status}, started_at=${startedJob.started_at}`,
    });

    // Test 4: Update progress
    await updateJobProgress(supabase, job.id, 50);
    const progressJob = await fetchJob(supabase, job.id);

    results.push({
      name: 'Job progress update',
      passed: progressJob?.progress === 50,
      details: `Progress=${progressJob?.progress}`,
    });

    // Test 5: Complete the job
    const completedJob = await completeJob(supabase, job.id);

    results.push({
      name: 'Job completion',
      passed: completedJob.status === 'completed' && completedJob.progress === 100,
      details: `Status=${completedJob.status}, progress=${completedJob.progress}`,
    });

    // Test 6: After completion, creating same target creates NEW job
    const testRelatedId2 = crypto.randomUUID();
    const { job: newJob1 } = await createJob(supabase, {
      agencyId,
      jobType: 'llm_generation',
      relatedType: 'test',
      relatedId: testRelatedId2,
    });
    // Fail it
    await startJob(supabase, newJob1.id);
    await failJob(supabase, newJob1.id, 'Test failure');

    const canRetry = await canRetryJob(supabase, 'test', testRelatedId2, 'llm_generation');

    results.push({
      name: 'Retry control',
      passed: canRetry === true,
      details: `canRetry=${canRetry} (1 failure, max=3)`,
    });

  } catch (err) {
    results.push({
      name: 'Job lifecycle',
      passed: false,
      details: `Error: ${err instanceof Error ? err.message : String(err)}`,
    });
  }

  // Test 7: LLM Gateway (requires Edge Function deployed)
  try {
    const response = await generateContent(supabase, {
      type: 'alert_recommendation',
      variables: {
        alert_domain: 'profitability_collaborator',
        indicator_key: 'cost_ca_ratio',
        measured_value: '115%',
        threshold_value: '100%',
        severity: 'critical',
        agency_context: 'Agence immobilière de 5 collaborateurs, CA mensuel moyen 40K€',
      },
      agencyId,
    });

    results.push({
      name: 'LLM Gateway call',
      passed: response !== null && response.content.length > 0,
      details: response
        ? `Content length=${response.content.length}, generationId=${response.generationId}`
        : 'Returned null (LLM unavailable — graceful degradation OK)',
    });
  } catch (err) {
    results.push({
      name: 'LLM Gateway call',
      passed: false,
      details: `Error: ${err instanceof Error ? err.message : String(err)}`,
    });
  }

  return results;
}
```

// src/app/api/v3-test/route.ts
// Temporary test endpoint for V3.1 validation — remove after testing
import { NextResponse } from 'next/server';
import { createClient } from '@/lib/supabase/server';
import { runV31IntegrationTests } from '@/features/shared/services/__tests__/job-orchestrator.test';

export async function GET() {
  const supabase = await createClient();

  // Get current user's agency
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) {
    return NextResponse.json({ error: 'Not authenticated' }, { status: 401 });
  }

  const { data: membership } = await supabase
    .from('agency_members')
    .select('agency_id')
    .eq('user_id', user.id)
    .limit(1)
    .single();

  if (!membership) {
    return NextResponse.json({ error: 'No agency found' }, { status: 404 });
  }

  const results = await runV31IntegrationTests(supabase, membership.agency_id);

  const allPassed = results.every((r) => r.passed);

  return NextResponse.json({
    status: allPassed ? 'ALL PASSED' : 'SOME FAILED',
    results,
  });
}
